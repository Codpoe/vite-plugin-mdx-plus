# CodeBlock

## Nomral

```tsx
export interface Props {
  name: string;
}

export function CodeBlock({ name }: Props) {
  return <div>Hello {name}</div>
}
```

```sh
pnpm add -D vite-plugin-mdx-plus
```

## Diff

````md
```ts
console.log('hewwo') // [\!code --]

console.log('goodbye')
```
````
